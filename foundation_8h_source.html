149 static inline void key_idx_pack(struct os_mbuf *buf,
150  u16_t idx1, u16_t idx2)
151 {
152  net_buf_simple_add_le16(buf, idx1 | ((idx2 & 0x00f) << 12));
153  net_buf_simple_add_u8(buf, idx2 >> 4);
154 }

156 static inline void key_idx_unpack(struct os_mbuf *buf,
157  u16_t *idx1, u16_t *idx2)
158 {
159  *idx1 = sys_get_le16(&buf->om_data[0]) & 0xfff;
160  *idx2 = sys_get_le16(&buf->om_data[1]) >> 4;
161  net_buf_simple_pull(buf, 3);
162 }
