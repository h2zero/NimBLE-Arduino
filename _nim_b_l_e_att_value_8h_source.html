41/* Used to determine if the type passed to a template has a data() and size() method. */
42template <typename T, typename = void, typename = void>

45template <typename T>

229 template <typename T>
230 typename std::enable_if<!std::is_pointer<T>::value, bool>::type
231 setValue(const T& s) {
232 if constexpr (Has_data_size<T>::value) {
233 return setValue(reinterpret_cast<const uint8_t*>(s.data()), s.size());
234 } else if constexpr (Has_c_str_length<T>::value) {
235 return setValue(reinterpret_cast<const uint8_t*>(s.c_str()), s.length());
236 } else {
237 return setValue(reinterpret_cast<const uint8_t*>(&s), sizeof(s));
238 }
239 }
